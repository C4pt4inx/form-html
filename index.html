#------- [HttpPost] public ActionResult Form(ML.Usuario usuario) { HttpPostedFileBase file = RequestCacheLevel.Files["img"]; if(file.ContentLength > 0) { usuario.Imagen = ConvertBase64(file); } } public string ConvertBase64(HttpPostedFileBase Foto) { System.IO.BinaryReader reader = new System.IO.BinaryReader(Foto.InputStream); byte[] data = reader.ReadBytes((int)Foto.ContentLength); string imagen = Convert.ToBase64String(data); return imagen; }
